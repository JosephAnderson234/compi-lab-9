fun int fib(int n)
 var int a;
 var int b;
 var int c;
 var int i;
 a = 0;
 b = 1;
 if (n <= 0) then
   return(0)
 endif;
 if (n == 1) then
   return(1)
 endif;
 i = 2;
 while (i <= n) do
   c = a + b;
   a = b;
   b = c;
   i = i + 1
 endwhile;
 return(b)
endfun
